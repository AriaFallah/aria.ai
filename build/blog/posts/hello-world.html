function memoize<T: Function>(f: T): T {
  const cache = {};

  function m(...args) {
    const k = JSON.stringify(args);
    if (cache[k]) {
      return cache[k];
    }
    const result = f(...args);
    cache[k] = result;
    return result;
  }

  return ((m: any): T);
}
